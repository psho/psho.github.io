    Select Rows¶

    By column value¶

    Must use bitwise operators '&' (AND) and '|' (OR)

    In [ ]:
    # Select rows on condition
df[df['col'] == 5]
df[(df['col'] > 5) & (df['col'] < 10)]

# Select rows with nulls in specified column
df[df['col'].isnull()]

# Select rows without nulls in specified column
df[df['col'].notnull()]
    By index label¶

    Selecting rows by index is faster than '=='. Even faster if already sorted by index.

    In [ ]:
    df.loc[0]          # row with index label 0
df.loc[1:3]        # rows with index label 1 to 3 inclusive
df.loc['A']        # rows with index label 'A'
df.loc[['A','B']]  # rows with index label 'A' or 'B'
df.loc['A', 15]    # rows where first index is 'A' and second index is 15
    By position¶
    In [ ]:
    df.iloc[0]         # first row
df.iloc[1:5]       # rows at position 1 to 4

    Select Columns and Rows¶

    In [ ]:
    # col1 and col2, rows where col3<4
df[['col1','col2']][df['col3']<4]

# col1 and col2, rows with index label 'A' or 'B'
df[['col1','col2']].loc[['A', 'B']]

# col1 and col2, rows at position 1 to 3
df[['col1','col2']].iloc[1:4]

    Append¶

    In [ ]:
    # Keep duplicate rows
pd.concat([df1, df2])

# Drop duplicate rows
pd.concat([df1, df2]).drop_duplicates()

# Drop duplicate rows based on specified columns, keeping first occurrence
pd.concat([df1, df2]).drop_duplicates(subset=['col1', 'col2'])

# Drop duplicate rows based on specified columns, keeping last occurrence
pd.concat([df1, df2]).drop_duplicates(subset=['col1', 'col2'], keep='last')

    Join¶

    In [ ]:
    # Column to column inner join
pd.merge(df1, df2,
         left_on='coll', right_on='colr',
         how='inner',
         suffixes=['', '_r'])  # column name suffixes for duplicates

# Index to index left join
pd.merge(df1, df2,
         left_index=True, right_index=True,
         how='left',
         suffixes=['', '_r'])

# Column to index outer join
pd.merge(df1, df2,
         left_on='coll', right_index=True,
         how='outer',
         suffixes=['', '_r'])

    Stacking/Unstacking¶

    Dataframes require both left and top index. If you stack/unstack too much, it will become a Series.

    In [72]:
    df = pd.DataFrame({'foo': ['one', 'one', 'two', 'two'],
                   'bar': ['A', 'B', 'A', 'B'],
                   'baz': [1, 3, 5, 7],
                   'zoo': [2, 4, 6, 8]}).set_index(['foo', 'bar'])
df
    Out[72]:
    baz zoo
    foo bar
    one A 1 2
    B 3 4
    two A 5 6
    B 7 8
    In [74]:
    # Move innermost left index 'up' to innermost top index - Wider
df_unstacked = df.unstack()
df_unstacked
    Out[74]:
    baz zoo
    bar A B A B
    foo
    one 1 3 2 4
    two 5 7 6 8
    In [75]:
    # Move innermost top index 'down' to innermost left index - Taller
df_unstacked.stack()
    Out[75]:
    baz zoo
    foo bar
    one A 1 2
    B 3 4
    two A 5 6
    B 7 8

    Reading¶

    • 'r' reading
    • 'r+' reading and writing
    • 'rb' read binary file
    In [ ]:
    # Reading line by line
f = open('file.txt', 'r')

line1 = f.readline()  # read a line, append \n, move pointer to next line
line2 = f.readline()
line3 = f.readline()

f.close()


# Reading lines with for loop
f = open('file.txt', 'r')

for line in f:
    print(line)

f.close()


# Reading files with set buffer size
f = open('file.txt', 'r')

while len(buf)>0:
    buf = f.read(20)  # 20 bytes at a time
    
f.close()

    Main Function¶

    • If running script as main:
      • __name__ will be set to __main__
      • Following code will execute
    • If importing as module:
      • __name__ will be something else
      • Following code will not execute
    In [ ]:
    # Basic
def main():
    # code

if __name__ == "__main__":
    main()
    In [ ]:
    # With arguments - argv[0] refers to python script, argv[1] is first argument etc.
import sys

def main(argv):
    print(str(argv))

if __name__ == "__main__":
    main(sys.argv)

    Create¶

    In [ ]:
    a_string = 'abc'
b_string = 'xyz'

# Convert to string
c_string = str(2.1)

    Inspect¶

    In [ ]:
    # Number of characters in string
len(a_string)

# Check if all characters...
a_string.isalpha()  # are alphabetic
a_string.isdigit()  # are digits
a_string.isalnum()  # are alphanumeric
a_string.islower()  # are lowercase, ignores digits/punctuation
a_string.isupper()  # are uppercase, ignores digits/punctuation
a_string.isspace()  # are whitespaces, includes \n and \t
a_string.istitle()  # are titlecased, i.e. first letter of each word is cap

# Unique strings only
string_list.unique()

    Find substrings¶

    In [ ]:
    # Number of substring occurrences
a_string.count('sub')

# Position of substring
a_string.index('sub')  # returns ValueError if not found
a_string.find('sub')   # same, but returns -1 if not found

# Check if starts/ends with substring
a_string.startswith('sub')
a_string.startswith(('foo', 'bar'))  # can specify tuple of possible matches
a_string.endswith('sub')

# Replace substrings
a_string.replace('old', 'new')
a_string.replace('old', 'new', 3)  # replace first 3 occurrences only

    Cleaning¶

    In [ ]:
    # Convert all characters to upper/lowercase
a_string.upper()
a_string.lower()

# Remove leading and trailing characters
a_string.strip()       # whitespace
a_string.strip('ab ')  # remove leading/trailing 'a', 'b', ' '

    Combining¶

    In [ ]:
    # Concatenate
a_string + ' ' + b_string

# Join up a list/tuple/string with specified separator in between
