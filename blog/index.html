    Python: Dictionaries

    Patrick Ho

    2018-07-27 23:16

    Create¶

    In [ ]:
    # Init empty dictionary
empty_dict = {}

# Init dictionary with key:value pairs
a_dict = {'Alice':30, 'Bob':35, 'Chris':28, 'Dave':'N/A'}

# Init dictionary using dict()
a_dict = dict(Alice=30, Bob=35, Chris=28, Dave='N/A')

    Inspect¶

    In [ ]:
    # Number of key-value pairs in dictionary
len(a_dict)

# Check if key is in dictionary
'a' in a_dict
any(x in a_dict for x in ['a', 'b'])  # check if any items in dictionary
all(x in a_dict for x in ['a', 'b'])  # check if all items in dictionary

# Check if value is in dictionary
'a' in a_dict.values()

# List of keys
a_dict.keys()

# List of values
a_dict.values()

# List of key-value pairs (as tuples)
a_dict.items()

    Access¶

    In [ ]:
    # Lookup single value by key
a_dict['Bob']
a_dict.get('Frank')               # returns None
a_dict.get('Frank', 'Not Found')  # returns 'Not Found'

    Add items to the dictionary¶

    In [ ]:
    # Add new item
a_dict['Ed'] = 49

    Edit items in the dictionary¶

    In [ ]:
    # Edit existing item
a_dict['Dave'] = 42

# Assignment creates reference, not a copy!
a_dict = {'a':1, 'b':2, 'c':3}
b_dict = a_dict
a_dict['a'] = 99  # both values for 'a' and 'b' will be 99

    Remove items from the dictionary¶

    In [ ]:
    # Delete item
del a_dict['Alice']

# Remove all items leaving empty dictionary
a_dict.clear()

# Delete dictionary
del a_dict

    Sort¶

    Dictionaries have no order, can either:

    • Create a sorted representation as a list
    • Use OrderedDict
    In [ ]:
    import operator
import collections

# Sorted by value, represent as a list of tuples
a_dict = {'a':3, 'b':5, 'c':2, 'd':8}
a_dict_sorted = sorted(a_dict.items(), key=operator.itemgetter(1))

# Store as OrderedDict
a_dict_ordered = collections.OrderedDict(a_dict_sorted)

    Combine¶

    In [ ]:
    # Add a dictionaries key-value pairs to another with dupes removed
a_dict.update(dict2)

    Python: Files

    Patrick Ho

    2018-07-27 23:16

    Reading¶

    • 'r' reading
    • 'r+' reading and writing
    • 'rb' read binary file
    In [ ]:
    # Reading line by line
f = open('file.txt', 'r')

line1 = f.readline()  # read a line, append \n, move pointer to next line
line2 = f.readline()
line3 = f.readline()

f.close()


# Reading lines with for loop
f = open('file.txt', 'r')

for line in f:
    print(line)

f.close()


# Reading files with set buffer size
f = open('file.txt', 'r')

while len(buf)>0:
    buf = f.read(20)  # 20 bytes at a time
    
f.close()

    Writing¶

    • 'w' writing, creates/overwrites
    • 'a' appending, creates/appends
    • 'wb' write binary file
    In [ ]:
    # Writing line by line

f.write('Line 1\n')
f.write('Line 2\n')

f.close()

    Functions¶

    • Variables defined within function are not available outside function
    • Variables within function overrides global variables of same name
    In [1]:

    result = val * 2
    return result

multiplyByTwo(3)

    Main Function¶

    • If running script as main:
      • __name__ will be set to __main__
      • Following code will execute
    • If importing as module:
      • __name__ will be something else
      • Following code will not execute
    In [ ]:
    # Basic
def main():
    # code

if __name__ == "__main__":
    main()
    In [ ]:
    # With arguments - argv[0] refers to python script, argv[1] is first argument etc.
import sys

def main(argv):
    print(str(argv))

if __name__ == "__main__":
    main(sys.argv)

    Python: Lists

    Patrick Ho

    2018-07-27 23:16

    Create¶

    In [ ]:
    # Init empty list
empty_list = []

# Init list with values
a_list = [5, 10, 15, 20, 25, 30, 35, 40, 45, 'dog']

    Inspect¶

    In [ ]:
    # Number of items in list
len(a_list)

# Check if item is in list
'a' in a_list
any(x in a_list for x in ['a', 'b'])  # check if any items in list
all(x in a_list for x in ['a', 'b'])  # check if all items in list

    Access¶

    In [ ]:
    # Access single item
a_list[0]      # first item
a_list[-2]     # second last item

# Access slice of items
a_list[:3]     # first 3 items
a_list[4:]     # 5th item until last
a_list[-3:]    # last 3 items
a_list[2:6]    # items at index 2 to 5 (6-1)
a_list[2:8:2]  # index steps of 2

    Adding to the list¶

    In [ ]:
    # Add item to end of list
a_list.append('a')

# Insert item to a specified index position
a_list.insert(3, 'a')

    Edit items in the list¶

    In [ ]:
    # Overwrite existing item
a_list[10] = 1000

# Assignment creates reference, not a copy!
a = [1,2,3]
b = a
a[0] = 99  # both a[0] and b[0] will be 99

    Remove items from the list¶

    In [ ]:
    # Remove item from list by index
del a_list[0]    # delete first item
del a_list[3:5]  # delete items at index 3 to 5

# Remove item from list by value
a_list.remove('a')

# Retrieve item from list and remove it
a_list.pop()   # last item
a_list.pop(2)  # item at index 2

    Sort¶

    In [ ]:
    # Sort list in-place
a_list.sort()              # default ascending
a_list.sort(reverse=True)  # descending

# Sort into new list and retain original unsorted list
new_list = sorted(orig_list)

# Reverse order
a_list.reverse()

    Combine¶

    In [ ]:
    # Concatenate
a_list + b_list
a_list.extend(b_list)  # alternative method, will update a_list

# Duplicate and append
a_list * 3  # i.e. a_list + a_list + a_list

    Python: Strings

    Patrick Ho

    2018-07-27 23:16

    Create¶

    In [ ]:
    a_string = 'abc'
b_string = 'xyz'

# Convert to string
c_string = str(2.1)

    Inspect¶

    In [ ]:
    # Number of characters in string
len(a_string)

# Check if all characters...
a_string.isalpha()  # are alphabetic
a_string.isdigit()  # are digits
a_string.isalnum()  # are alphanumeric
a_string.islower()  # are lowercase, ignores digits/punctuation
a_string.isupper()  # are uppercase, ignores digits/punctuation
a_string.isspace()  # are whitespaces, includes \n and \t
a_string.istitle()  # are titlecased, i.e. first letter of each word is cap

# Unique strings only
string_list.unique()

    Find substrings¶

    In [ ]:
    # Number of substring occurrences
a_string.count('sub')

# Position of substring
a_string.index('sub')  # returns ValueError if not found
a_string.find('sub')   # same, but returns -1 if not found

# Check if starts/ends with substring
a_string.startswith('sub')
a_string.startswith(('foo', 'bar'))  # can specify tuple of possible matches
a_string.endswith('sub')

# Replace substrings
a_string.replace('old', 'new')
a_string.replace('old', 'new', 3)  # replace first 3 occurrences only

    Cleaning¶

    In [ ]:
    # Convert all characters to upper/lowercase
a_string.upper()
a_string.lower()

# Remove leading and trailing characters
a_string.strip()       # whitespace
a_string.strip('ab ')  # remove leading/trailing 'a', 'b', ' '

    Combining¶

    In [ ]:
    # Concatenate
a_string + ' ' + b_string

# Join up a list/tuple/string with specified separator in between
sep = '-'
list1 = ['aaa', 'bbb', 'ccc']
a_string = 'Hello World!'
sep.join(list1)    # 'aaa-bbb-ccc'
sep.join(a_string)  # 'H-e-l-l-o- -W-o-r-l-d-!'

    Splitting¶

    In [ ]:
    # Split up string into a list of substrings
a_string.split()        # use space as separator
a_string.split(',')     # use comma as separator
a_string.split(' ', 2)  # use space as separator, only first 2 splits

# Split up string by lines
a_string.splitlines()               # use line breaks as separator
a_string.splitlines(keepends=True)  # keep line break characters
