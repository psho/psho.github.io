Set Tick Labels¶

In [8]:
plt.plot(x, y_cos)
plt.plot(x, y_sin)

ax = plt.gca()  # get current axis
for label in ax.get_xticklabels() + ax.get_yticklabels():
    label.set_fontsize(12)
    label.set_bbox(dict(facecolor='red', edgecolor='None', alpha=0.65))

Set Spines¶

Lines connecting the axis tick marks and forming boundaries of the data area: top/bottom/left/right

In [31]:
plt.plot(x, y_cos)
plt.plot(x, y_sin)

# Get current axes
ax = plt.gca()

# Hide top and right spines
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)

# Move left spine to where x=0 and show ticks on left
ax.spines['left'].set_position(('data',0))
ax.yaxis.set_ticks_position('left')

# Move bottom spine to where y=0 and show ticks below
ax.spines['bottom'].set_position(('data',0))
ax.xaxis.set_ticks_position('bottom')

Annotate Points ¶

In [34]:
plt.plot(x, y_cos)
plt.plot(x, y_sin)

# Set x and y coordinates of point to annotate
x_note = 2*np.pi/3
y_note = np.cos(x_note)

# Mark point
plt.scatter([x_note,], [y_note,],
            s=50, color='blue')

# Draw a dotted line from y=0 to the point
plt.plot([x_note, x_note], [0, y_note],
         color='blue', linewidth=1.5, linestyle="dotted")

# Annotate point with text offset to bottom left and draw curved arrow 
plt.annotate(r'$\cos(\frac{2\pi}{3})=-\frac{1}{2}$',
             xy=(x_note, y_note), xycoords='data',  
             xytext=(-90, -50), textcoords='offset points',
             fontsize=16,
             arrowprops=dict(arrowstyle="->", connectionstyle="arc3,rad=.2"))
Out[34]:
