Bar Chart¶

In [46]:
labels = ['a', 'b', 'c', 'd', 'e']
values = [1, 3, 2, 4, 3]

# Set x axis tick locations
x_locs = [i for i, _ in enumerate(labels)]
In [47]:
plt.bar(x_locs, values)

plt.xlabel('x axis')
plt.ylabel('y axis')
plt.title('chart title')
plt.xticks(x_locs, labels);
