    Scatterplots¶

    In [11]:
    x_values = [4, 6, 7, 11, 17, 18, 18, 26, 39, 43]
y_values = [102, 106, 126, 133, 133, 143, 151, 152, 175, 177]
labels   = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j']
    In [12]:
    plt.scatter(x_values, y_values, color='red')

# label points
for label, x_pos, y_pos in zip(labels, x_values, y_values):
    plt.annotate(label,
                 xy=(x_pos, y_pos),
                 xytext=(5, -5),
                 textcoords='offset points');

    Equal Axes¶

    If units are the same, the default scale may be misleading. Use equal scale axes for a better representation.

    In [13]:
    z_values = [18, 19, 33, 37, 71, 79, 85, 89, 95, 96]
    In [14]:
    plt.scatter(x_values, z_values);
    In [15]:
    plt.scatter(x_values, z_values)
plt.axis('equal');
